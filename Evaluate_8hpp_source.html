26 template <size_t NumIndices>
27 constexpr bool contains_indices_to_contract(
28  const std::array<size_t, NumIndices>& tensorindices) noexcept {
29  if constexpr (NumIndices < 2) {
30  return false;
31  } else {
32  for (size_t i = 0; i < NumIndices - 1; i++) {
33  for (size_t j = i + 1; j < NumIndices; j++) {
34  if (gsl::at(tensorindices, i) ==
35  get_tensorindex_value_with_opposite_valence(
36  gsl::at(tensorindices, j))) {
37  return true;
38  }
39  }
40  }
41  return false;
42  }

45 /// \brief Helper struct for checking that a RHS tensor's index can be evaluated
46 /// to its corresponding index in the LHS tensor
47 ///
48 /// \details
49 /// A RHS index's corresponding LHS index uses the same generic index, such as
50 /// `ti_a`. For it to be possible to evaluate a RHS index to its corresponding
51 /// LHS index, this checks that the following is true for the index on both
52 /// sides:
53 /// - has the same valence (`UpLo`)
54 /// - has the same `Frame` type
55 /// - has the same number of spatial dimensions (allowing for expressions that
56 /// use generic spatial indices for spacetime indices on either side)
57 ///
58 /// \tparam LhsIndexList the LHS tensor's \ref SpacetimeIndex "TensorIndexType"
59 /// list
60 /// \tparam RhsIndexList the RHS tensor's \ref SpacetimeIndex "TensorIndexType"
61 /// list
62 /// \tparam LhsTensorIndexList the LHS tensor's generic index list
63 /// \tparam RhsTensorIndexList the RHS tensor's generic index list
64 /// \tparam CurrentLhsTensorIndex the current generic index of the LHS tensor
65 /// that is being checked, e.g. the type of `ti_a`
66 template <typename LhsIndexList, typename RhsIndexList,
67  typename LhsTensorIndexList, typename RhsTensorIndexList,
68  typename CurrentLhsTensorIndex>
69 struct EvaluateIndexCheckHelper {
70  using lhs_index =
71  tmpl::at<LhsIndexList,
72  tmpl::index_of<LhsTensorIndexList, CurrentLhsTensorIndex>>;
73  using rhs_index =
74  tmpl::at<RhsIndexList,
75  tmpl::index_of<RhsTensorIndexList, CurrentLhsTensorIndex>>;
76 
77  using type = std::bool_constant<
78  lhs_index::ul == rhs_index::ul and
79  std::is_same_v<typename lhs_index::Frame, typename rhs_index::Frame> and
80  ((lhs_index::index_type == rhs_index::index_type and
81  lhs_index::dim == rhs_index::dim) or
82  (lhs_index::index_type == IndexType::Spacetime and
83  lhs_index::dim == rhs_index::dim + 1) or
84  (rhs_index::index_type == IndexType::Spacetime and
85  lhs_index::dim + 1 == rhs_index::dim))>;
86 };

112 /// \brief Computes a transformation from the LHS tensor's multi-indices to
113 /// the equivalent RHS tensor's multi-indices, according to the differences in
114 /// the orderings of their generic indices
115 ///
116 /// \details
117 /// The elements of the transformation are the positions of the RHS generic
118 /// indices in the LHS generic indices. Put another way, for some `i`,
119 /// `rhs_tensorindices[i] == lhs_tensorindices[index_transformation[i]]`.
120 ///
121 /// Here is an example of what the algorithm does:
122 ///
123 /// Tensor equation: \f$L_{cab} = R_{abc}\f$
124 /// `lhs_tensorindices`:
125 /// \code
126 /// {2, 0, 1} // i.e. {c, a, b}
127 /// \endcode
128 /// `rhs_tensorindices`:
129 /// \code
130 /// {0, 1, 2} // i.e. {a, b, c}
131 /// \endcode
132 /// returned `index_transformation`:
133 /// \code
134 /// {1, 2, 0} // positions of RHS indices {a, b, c} in LHS indices {c, a, b}
135 /// \endcode
136 ///
137 /// \tparam NumIndices the number of indices in the tensors
138 /// \param lhs_tensorindices the TensorIndexs of the LHS tensor
139 /// \param rhs_tensorindices the TensorIndexs of the RHS tensor
140 /// \return a transformation from the LHS tensor's multi-indices to the
141 /// equivalent RHS tensor's multi-indices
142 template <size_t NumIndices>
143 SPECTRE_ALWAYS_INLINE constexpr std::array<size_t, NumIndices>
144 compute_index_transformation(
145  const std::array<size_t, NumIndices>& lhs_tensorindices,
146  const std::array<size_t, NumIndices>& rhs_tensorindices) noexcept {
147  std::array<size_t, NumIndices> index_transformation{};
148  for (size_t i = 0; i < NumIndices; i++) {
149  gsl::at(index_transformation, i) = static_cast<size_t>(std::distance(
150  lhs_tensorindices.begin(),
151  alg::find(lhs_tensorindices, gsl::at(rhs_tensorindices, i))));
152  }
153  return index_transformation;
154 }

156 /// \brief Computes the RHS tensor multi-index that is equivalent to a given
157 /// LHS tensor multi-index, according to the differences in the orderings of
158 /// their generic indices
159 ///
160 /// \details
161 /// Here is an example of what the algorithm does:
162 ///
163 /// Tensor equation: \f$L_{cab} = R_{abc}\f$
164 /// `index_transformation`:
165 /// \code
166 /// {1, 2, 0} // positions of RHS indices {a, b, c} in LHS indices {c, a, b}
167 /// \endcode
168 /// `lhs_multi_index`:
169 /// \code
170 /// {3, 4, 5} // i.e. c = 3, a = 4, b = 5
171 /// \endcode
172 /// returned equivalent `rhs_multi_index`:
173 /// \code
174 /// {4, 5, 3} // i.e. a = 4, b = 5, c = 3
175 /// \endcode
176 ///
177 /// \tparam NumIndices the number of indices in the tensors
178 /// \param lhs_multi_index the multi-index of the LHS tensor
179 /// \param index_transformation the list of the positions of the RHS indices in
180 /// the LHS indices
181 /// \return the RHS tensor multi-index that is equivalent to `lhs_tensor_index`
182 template <size_t NumIndices>
183 SPECTRE_ALWAYS_INLINE constexpr std::array<size_t, NumIndices>
184 compute_rhs_multi_index(
185  const std::array<size_t, NumIndices>& lhs_multi_index,
186  const std::array<size_t, NumIndices>& index_transformation) noexcept {
187  std::array<size_t, NumIndices> rhs_multi_index{};
188  for (size_t i = 0; i < NumIndices; i++) {
189  gsl::at(rhs_multi_index, i) =
190  gsl::at(lhs_multi_index, gsl::at(index_transformation, i));
191  }
192  return rhs_multi_index;
193 }

195 /*!
196  * \ingroup TensorExpressionsGroup
197  * \brief Evaluate a RHS tensor expression to a tensor with the LHS index order
198  * set in the template parameters
199  *
200  * \details Uses the right hand side (RHS) TensorExpression's index ordering
201  * (`RhsTE::args_list`) and the desired left hand side (LHS) tensor's index
202  * ordering (`LhsTensorIndices`) to fill the provided LHS Tensor with that LHS
203  * index ordering. This can carry out the evaluation of a RHS tensor expression
204  * to a LHS tensor with the same index ordering, such as \f$L_{ab} = R_{ab}\f$,
205  * or different ordering, such as \f$L_{ba} = R_{ab}\f$.
206  *
207  * The symmetry of the provided LHS Tensor need not match the symmetry
208  * determined from evaluating the RHS TensorExpression according to its order of
209  * operations. This allows one to specify LHS symmetries (via `lhs_tensor`) that
210  * may not be preserved by the RHS expression's order of operations, which
211  * depends on how the expression is written and implemented.
212  *
213  * ### Example usage
214  * Given two rank 2 Tensors `R` and `S` with index order (a, b), add them
215  * together and fill the provided resultant LHS Tensor `L` with index order
216  * (b, a):
217  * \code{.cpp}
218  * TensorExpressions::evaluate<ti_b, ti_a>(
219  * make_not_null(&L), R(ti_a, ti_b) + S(ti_a, ti_b));
220  * \endcode
221  *
222  * This represents evaluating: \f$L_{ba} = R_{ab} + S_{ab}\f$
223  *
224  * Note: `LhsTensorIndices` must be passed by reference because non-type
225  * template parameters cannot be class types until C++20.
226  *
227  * @tparam LhsTensorIndices the TensorIndexs of the Tensor on the LHS of the
228  * tensor expression, e.g. `ti_a`, `ti_b`, `ti_c`
229  * @param lhs_tensor pointer to the resultant LHS Tensor to fill
230  * @param rhs_tensorexpression the RHS TensorExpression to be evaluated
231  */
232 template <auto&... LhsTensorIndices, typename X, typename LhsSymmetry,
233  typename LhsIndexList, typename Derived, typename RhsSymmetry,
234  typename RhsIndexList, typename... RhsTensorIndices>
235 void evaluate(
236  const gsl::not_null<Tensor<X, LhsSymmetry, LhsIndexList>*> lhs_tensor,
237  const TensorExpression<Derived, X, RhsSymmetry, RhsIndexList,
238  tmpl::list<RhsTensorIndices...>>&
239  rhs_tensorexpression) {
240  using lhs_tensorindex_list =
241  tmpl::list<std::decay_t<decltype(LhsTensorIndices)>...>;
242  using rhs_tensorindex_list = tmpl::list<RhsTensorIndices...>;
243  static_assert(
244  tmpl::equal_members<lhs_tensorindex_list, rhs_tensorindex_list>::value,
245  "The generic indices on the LHS of a tensor equation (that is, the "
246  "template parameters specified in evaluate<...>) must match the generic "
247  "indices of the RHS TensorExpression. This error occurs as a result of a "
248  "call like evaluate<ti_a, ti_b>(R(ti_A, ti_b) * S(ti_a, ti_c)), where "
249  "the generic indices of the evaluated RHS expression are ti_b and ti_c, "
250  "but the generic indices provided for the LHS are ti_a and ti_b.");
251  static_assert(
252  tmpl::is_set<std::decay_t<decltype(LhsTensorIndices)>...>::value,
253  "Cannot evaluate a tensor expression to a LHS tensor with a repeated "
254  "generic index, e.g. evaluate<ti_a, ti_a>.");
255  static_assert(
256  not detail::contains_indices_to_contract<sizeof...(LhsTensorIndices)>(
257  {{std::decay_t<decltype(LhsTensorIndices)>::value...}}),
258  "Cannot evaluate a tensor expression to a LHS tensor with generic "
259  "indices that would be contracted, e.g. evaluate<ti_A, ti_a>.");
260  // `EvaluateIndexCheck` does also check that valence (Up/Lo) of indices that
261  // correspond in the RHS and LHS tensors are equal, but the assertion message
262  // below does not mention this because a mismatch in valence should have been
263  // caught due to the combination of (i) the Tensor::operator() assertion
264  // checking that generic indices' valences match the tensor's indices'
265  // valences and (ii) the above assertion that RHS and LHS generic indices
266  // match
267  static_assert(
268  detail::EvaluateIndexCheck<LhsIndexList, RhsIndexList,
269  lhs_tensorindex_list,
270  rhs_tensorindex_list>::value,
271  "At least one index of the tensor evaluated from the RHS expression "
272  "cannot be evaluated to its corresponding index in the LHS tensor. This "
273  "is due to a difference in number of spatial dimensions or Frame type "
274  "between the index on the RHS and LHS. "
275  "e.g. evaluate<ti_a, ti_b>(L, R(ti_b, ti_a));, where R's first "
276  "index has 2 spatial dimensions but L's second index has 3 spatial "
277  "dimensions. Check RHS and LHS indices that use the same generic index.");
278 
279  using lhs_tensor_type = typename std::decay_t<decltype(*lhs_tensor)>;
280 
281  // positions of indices in LHS tensor where generic spatial indices are used
282  // for spacetime indices
283  constexpr auto lhs_spatial_spacetime_index_positions =
284  detail::get_spatial_spacetime_index_positions<LhsIndexList,
285  lhs_tensorindex_list>();
286  // positions of indices in RHS tensor where generic spatial indices are used
287  // for spacetime indices
288  constexpr auto rhs_spatial_spacetime_index_positions =
289  detail::get_spatial_spacetime_index_positions<RhsIndexList,
290  rhs_tensorindex_list>();
291 
292  for (size_t i = 0; i < lhs_tensor_type::size(); i++) {
293  if constexpr (lhs_spatial_spacetime_index_positions.size() == 0) {
294  // either:
295  // (i) RHS nor LHS uses a generic spatial index for a spacetime index, or
296  // (ii) only RHS uses a generic spatial index for a spacetime index
297  auto rhs_multi_index = compute_rhs_multi_index(
298  lhs_tensor_type::structure::get_canonical_tensor_index(i),
299  compute_index_transformation<sizeof...(RhsTensorIndices)>(
300  {{std::decay_t<decltype(LhsTensorIndices)>::value...}},
301  {{RhsTensorIndices::value...}}));
302  for (size_t j = 0; j < rhs_spatial_spacetime_index_positions.size();
303  j++) {
304  gsl::at(rhs_multi_index,
305  gsl::at(rhs_spatial_spacetime_index_positions, j)) += 1;
306  }
307 
308  (*lhs_tensor)[i] =
309  (~rhs_tensorexpression)
310  .template get<RhsTensorIndices...>(rhs_multi_index);
311 
312  } else {
313  // either:
314  // (i) only LHS uses a generic spatial index for a spacetime index
315  // (ii) both RHS and LHS use a generic spatial index for a spacetime index
316  auto lhs_multi_index =
317  lhs_tensor_type::structure::get_canonical_tensor_index(i);
318  // Only evaluate the component at `lhs_multi_index` if it does not contain
319  // the time index (0) for any spacetime indices for which a generic
320  // spatial index is being used
321  if (alg::none_of(lhs_spatial_spacetime_index_positions,
322  [lhs_multi_index](size_t j) {
323  return gsl::at(lhs_multi_index, j) == 0;
324  })) {
325  for (size_t j = 0; j < lhs_spatial_spacetime_index_positions.size();
326  j++) {
327  gsl::at(lhs_multi_index,
328  gsl::at(lhs_spatial_spacetime_index_positions, j)) -= 1;
329  }
330  auto rhs_multi_index = compute_rhs_multi_index(
331  lhs_multi_index,
332  compute_index_transformation<sizeof...(RhsTensorIndices)>(
333  {{std::decay_t<decltype(LhsTensorIndices)>::value...}},
334  {{RhsTensorIndices::value...}}));
335  for (size_t j = 0; j < rhs_spatial_spacetime_index_positions.size();
336  j++) {
337  gsl::at(rhs_multi_index,
338  gsl::at(rhs_spatial_spacetime_index_positions, j)) += 1;
339  }
340 
341  (*lhs_tensor)[i] =
342  (~rhs_tensorexpression)
343  .template get<RhsTensorIndices...>(rhs_multi_index);
344  }
345  }
346  }
347 }

349 /*!
350  * \ingroup TensorExpressionsGroup
351  * \brief Evaluate a RHS tensor expression to a tensor with the LHS index order
352  * set in the template parameters
353  *
354  * \details Uses the right hand side (RHS) TensorExpression's index ordering
355  * (`RhsTE::args_list`) and the desired left hand side (LHS) tensor's index
356  * ordering (`LhsTensorIndices`) to construct a LHS Tensor with that LHS index
357  * ordering. This can carry out the evaluation of a RHS tensor expression to a
358  * LHS tensor with the same index ordering, such as \f$L_{ab} = R_{ab}\f$, or
359  * different ordering, such as \f$L_{ba} = R_{ab}\f$.
360  *
361  * The symmetry of the returned LHS Tensor depends on the order of operations in
362  * the RHS TensorExpression, i.e. how the expression is written. If you would
363  * like to specify the symmetry of the LHS Tensor instead of it being determined
364  * by the order of operations in the RHS expression, please use the other
365  * `evaluate` overload that takes an empty LHS Tensor as its first argument.
366  *
367  * ### Example usage
368  * Given two rank 2 Tensors `R` and `S` with index order (a, b), add them
369  * together and generate the resultant LHS Tensor `L` with index order (b, a):
370  * \code{.cpp}
371  * auto L = TensorExpressions::evaluate<ti_b, ti_a>(
372  * R(ti_a, ti_b) + S(ti_a, ti_b));
373  * \endcode
374  * \metareturns Tensor
375  *
376  * This represents evaluating: \f$L_{ba} = R_{ab} + S_{ab}\f$
377  *
378  * Note: If a generic spatial index is used for a spacetime index in the RHS
379  * tensor, its corresponding index in the LHS tensor type will be a spatial
380  * index with the same valence, frame, and number of spatial dimensions.
381  *
382  * Note: `LhsTensorIndices` must be passed by reference because non-type
383  * template parameters cannot be class types until C++20.
384  *
385  * @tparam LhsTensorIndices the TensorIndexs of the Tensor on the LHS of the
386  * tensor expression, e.g. `ti_a`, `ti_b`, `ti_c`
387  * @param rhs_tensorexpression the RHS TensorExpression to be evaluated
388  * @return the resultant LHS Tensor with index order specified by
389  * LhsTensorIndices
390  */
391 template <auto&... LhsTensorIndices, typename RhsTE,
392  Requires<std::is_base_of_v<Expression, RhsTE>> = nullptr>
393 auto evaluate(const RhsTE& rhs_tensorexpression) {
394  using lhs_tensorindex_list =
395  tmpl::list<std::decay_t<decltype(LhsTensorIndices)>...>;
396  using rhs_tensorindex_list = typename RhsTE::args_list;
397  using rhs_symmetry = typename RhsTE::symmetry;
398  using rhs_tensorindextype_list = typename RhsTE::index_list;
399 
400  // Stores (potentially reordered) symmetry and indices needed for constructing
401  // the LHS tensor, with index order specified by LhsTensorIndices
402  using lhs_tensor_symm_and_indices =
403  LhsTensorSymmAndIndices<rhs_tensorindex_list, lhs_tensorindex_list,
404  rhs_symmetry, rhs_tensorindextype_list>;
405 
406  Tensor<typename RhsTE::type, typename lhs_tensor_symm_and_indices::symmetry,
407  typename lhs_tensor_symm_and_indices::tensorindextype_list>
408  lhs_tensor{};
409  evaluate<LhsTensorIndices...>(make_not_null(&lhs_tensor),
410  rhs_tensorexpression);
411  return lhs_tensor;
412 }
