30 /*!
31  * \ingroup InitializationGroup
32  * \brief Initialize items related to the basic structure of the element
33  *
34  * DataBox:
35  * - Uses:
36  * - `domain::Tags::Domain<Dim, Frame::Inertial>`
37  * - `domain::Tags::InitialExtents<Dim>`
38  * - `domain::Tags::InitialRefinementLevels<Dim>`
39  * - Adds:
40  * - `domain::Tags::Mesh<Dim>`
41  * - `domain::Tags::Element<Dim>`
42  * - `domain::Tags::ElementMap<Dim, Frame::Inertial>`
43  * - `domain::Tags::Coordinates<Dim, Frame::Logical>`
44  * - `domain::Tags::Coordinates<Dim, Frame::Inertial>`
45  * - `domain::Tags::InverseJacobian<Dim, Frame::Logical, Frame::Inertial>`
46  * - `domain::Tags::DetInvJacobian<Frame::Logical, Frame::Inertial>`
47  *
48  * \note This action relies on the `SetupDataBox` aggregated initialization
49  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
50  * phase action list prior to this action.
51  */
52 template <size_t Dim>
53 struct InitializeDomain {
54  private:
55  using InitializeGeometry = elliptic::dg::InitializeGeometry<Dim>;
56 
57  public:
58  using initialization_tags =
59  tmpl::list<domain::Tags::InitialExtents<Dim>,
60  domain::Tags::InitialRefinementLevels<Dim>>;
61  using simple_tags = typename InitializeGeometry::return_tags;
62  using compute_tags = tmpl::list<>;
63 
64  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
65  typename ActionList, typename ParallelComponent>
66  static std::tuple<db::DataBox<DbTagsList>&&> apply(
67  db::DataBox<DbTagsList>& box,
68  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
69  const Parallel::GlobalCache<Metavariables>& /*cache*/,
70  const ElementId<Dim>& element_id, const ActionList /*meta*/,
71  const ParallelComponent* const /*meta*/) noexcept {
72  db::mutate_apply<typename InitializeGeometry::return_tags,
73  typename InitializeGeometry::argument_tags>(
74  InitializeGeometry{}, make_not_null(&box), element_id);
75  return {std::move(box)};
76  }
77 };
78 } // namespace elliptic::dg::Actions
