23 /// \brief Convert a `DataVector` to a `boost::math::quaternion`
24 ///
25 /// \details To convert to a quaternion, a `DataVector` must have either 3 or 4
26 /// components. If it has 3 components, the quaternion will be constructed with
27 /// 0 scalar part while the vector part is the `DataVector`. If the `DataVector`
28 /// has 4 components, the quaternion is just the `DataVector` itself.
29 boost::math::quaternion<double> datavector_to_quaternion(
30  const DataVector& input) noexcept {
31  ASSERT(input.size() == 3 or input.size() == 4,
32  "To form a quaternion, a DataVector can either have 3 or 4 components "
33  "only. This DataVector has "
34  << input.size() << " components.");
35  if (input.size() == 3) {
36  return boost::math::quaternion<double>(0.0, input[0], input[1], input[2]);
37  } else {
38  return boost::math::quaternion<double>(input[0], input[1], input[2],
39  input[3]);
40  }
41 }
42 /// Normalize a `boost::math::quaternion`
43 template <typename T>
44 void normalize_quaternion(
45  const gsl::not_null<boost::math::quaternion<T>*> input) noexcept {
46  *input /= abs(*input);
