102 /// Initialize background quantities for the elliptic DG operator, possibly
103 /// including the metric necessary for normalizing face normals
104 template <size_t Dim, typename BackgroundFields>
105 struct InitializeBackground {
106  using return_tags =
107  tmpl::list<::Tags::Variables<BackgroundFields>,
108  domain::Tags::Faces<Dim, ::Tags::Variables<BackgroundFields>>>;
109  using argument_tags = tmpl::list<
110  domain::Tags::Coordinates<Dim, Frame::Inertial>, domain::Tags::Mesh<Dim>,
111  domain::Tags::InverseJacobian<Dim, Frame::Logical, Frame::Inertial>>;
112 
113  template <typename Background>
114  void operator()(
115  const gsl::not_null<Variables<BackgroundFields>*> background_fields,
116  const gsl::not_null<
117  DirectionMap<Dim, Variables<BackgroundFields>>*>
118  face_background_fields,
119  const tnsr::I<DataVector, Dim>& inertial_coords, const Mesh<Dim>& mesh,
120  const InverseJacobian<DataVector, Dim, Frame::Logical, Frame::Inertial>&
121  inv_jacobian,
122  const Background& background) const noexcept {
123  *background_fields = variables_from_tagged_tuple(background.variables(
124  inertial_coords, mesh, inv_jacobian, BackgroundFields{}));
125  ASSERT(mesh.quadrature(0) == Spectral::Quadrature::GaussLobatto,
126  "Only Gauss-Lobatto quadrature is currently implemented for "
127  "slicing background fields to faces.");
128  for (const auto& direction : Direction<Dim>::all_directions()) {
129  // Possible optimization: Only the background fields in the
130  // System::fluxes_computer::argument_tags are needed on internal faces.
131  data_on_slice(
132  make_not_null(&(*face_background_fields)[direction]),
133  *background_fields, mesh.extents(), direction.dimension(),
134  index_to_slice_at(mesh.extents(), direction));
135  }
136  }
137 };

139 /// Normalize face normals, possibly using a background metric. Set
140 /// `InvMetricTag` to `void` to normalize face normals with the Euclidean
141 /// magnitude.
142 template <size_t Dim, typename InvMetricTag>
143 struct NormalizeFaceNormal {
144  using return_tags =
145  tmpl::list<domain::Tags::FaceNormal<Dim>,
146  domain::Tags::UnnormalizedFaceNormalMagnitude<Dim>>;
147  using argument_tags =
148  tmpl::conditional_t<std::is_same_v<InvMetricTag, void>, tmpl::list<>,
149  tmpl::list<InvMetricTag>>;
150 
151  template <typename... InvMetric>
152  void operator()(
153  const gsl::not_null<tnsr::i<DataVector, Dim>*> face_normal,
154  const gsl::not_null<Scalar<DataVector>*> face_normal_magnitude,
155  const InvMetric&... inv_metric) const noexcept {
156  magnitude(face_normal_magnitude, *face_normal, inv_metric...);
157  for (size_t d = 0; d < Dim; ++d) {
158  face_normal->get(d) /= get(*face_normal_magnitude);
159  }
160  }
161 };
