27 /*!
28  * \brief An analytic solution representing a specialization of the radiative
29  * Robinson-Trautman solution described in \cite Derry1970fk.
30  *
31  * \details This solution is not quite analytic, in the sense that there is a
32  * single scalar field that must be evolved. Ultimately, it is a partial
33  * specialization of the Characteristic equations such that \f$J = 0\f$ and the
34  * evolution equations have been manipulated to give a time evolution equation
35  * for \f$e^{-2 \beta}\f$, which is equivalent to the Robinson-Trautman scalar
36  * \f$\omega_{\text{RT}}\f$ (denoted \f$W\f$ in \cite Derry1970fk -- we deviate
37  * from their notation because the symbol \f$W\f$ is already used elsewhere in
38  * the CCE system).
39  *
40  * \note The value of \f$\omega_{\text{RT}}\f$ should be real and near 1, which
41  * is imposed by the solution itself, any modes specified in the input file are
42  * treated as perturbations to the leading value of 1.0 for the
43  * Robinson-Trautman scalar \f$\omega_{\text{RT}}\f$.
44  *
45  * \note The use of substep time-steppers in conjunction with `RobinsonTrautman`
46  * analytic data is explictly forbidden by checks in
47  * `Cce::Actions::InitializeWorldtubeBoundary`. The reason is that any time the
48  * `RobinsonTrautman` receives a request for data at a time in the past of its
49  * current state, it must re-start its internal time stepper from the beginning
50  * of the evolution. This is a reasonable cost for multistep methods that only
51  * have non-ordered time steps during self-start, but will lead to
52  * \f$\mathcal O(N^2)\f$ internal steps in the `RobinsonTrautman` solution if a
53  * substep method is used, where \f$N\f$ is the number of steps performed by the
54  * substep method. If you find an unavoidable need to use `RobinsonTrautman`
55  * with a substep method, you will either need to only do so for very short
56  * evolutions, or need to write more sophisticated caching of the internal time
57  * step data.
58  */

176  /*!
177  * \brief Compute the spherical coordinate metric of the Robinson-Trautman
178  * solution generated by the time-evolved scalar \f$\omega_{\text{RT}}\f$.
179  *
180  * \details The spacetime metric of the Robinson-Trautman solution can be
181  * expressed as a specialization of the Bondi-Sachs metric (note the metric
182  * signature change as compared to equation (1.2) from \cite Derry1970fk)
183  *
184  * \f[
185  * ds^2 = -((r W + 1) \omega_{\text{RT}} - r^2 U \bar U) (dt - dr)^2
186  * - 2 \omega_{\text{RT}} (dt - dr) dr
187  * - 2 r^2 U^A q_{AB} dx^B (dt - dr) + r^2 q_{A B} dx^A dx^B,
188  * \f]
189  *
190  * where \f$q_{A B}\f$ represents the angular unit sphere metric, and the
191  * remaining Bondi-Sachs scalars and angular tensors are defined in terms of
192  * the Robinson-Trautman scalar \f$\omega_{\text{RT}}\f$
193  *
194  * \f{align*}{
195  * W &= \frac{1}{r}\left(\omega_{\text{RT}} + \eth \bar \eth
196  * \omega_{\text{RT}} - 1\right) - \frac{2}{r^2 \omega_{\text{RT}}^2}\\
197  * U &\equiv U^A q_A = \frac{\eth \omega_{\text{RT}}}{r}.
198  * \f}
199  * and \f$q_A\f$ is the angular dyad on the unit sphere.
200  *
201  * The angular part of the metric can be expressed in terms of the \f$U\f$
202  * scalar as
203  *
204  * \f{align*}{
205  * g_{u \theta} &= r^2 \Re U\\
206  * g_{u \phi} &= r^2 \Im U
207  * \f}
208  */
209  void spherical_metric(
210  gsl::not_null<
211  tnsr::aa<DataVector, 3, ::Frame::Spherical<::Frame::Inertial>>*>
212  spherical_metric,
213  size_t l_max, double time) const noexcept override;

215  /*!
216  * \brief Compute radial derivative of the spherical coordinate metric of the
217  * Robinson-Trautman solution generated by the time-evolved scalar
218  * \f$\omega_{\text{RT}}\f$.
219  *
220  * \details The radial derivative (at constant time t) of the
221  * Robinson-Trautman solution is obtained by differentiating the expressions
222  * from the documentation for `RobinsonTrautman::spherical_metric()`:
223  *
224  * \f{align*}{
225  * (\partial_r g_{a b} + \partial_t g_{a b}) dx^a dx^b =
226  * - (\omega_{\text{RT}} (r \partial_r W + W) - 2 r U \bar U
227  * - r^2 (\bar U\partial_r U + U \partial_r \bar U)) (dt - dr)^2
228  * - 2 r U^A q_{A B} dx^B (dt - dr) + 2 r q_{A B} dx^A dx^B
229  * \f}
230  *
231  * where \f$q_{A B}\f$ represents the angular unit sphere metric, and the
232  * remaining Bondi-Sachs scalars and angular tensors are defined in terms of
233  * the Robinson-Trautman scalar \f$\omega_{\text{RT}}\f$
234  *
235  * \f{align*}{
236  * W &= \frac{1}{r}\left(\omega_{\text{RT}}
237  * + \eth \bar \eth \omega_{\text{RT}} - 1\right)
238  * - \frac{2}{r^2 \omega_{\text{RT}}^2}\\
239  * \partial_r W &= -\frac{1}{r^2} \left(\omega_{\text{RT}}
240  * + \eth \bar \eth \omega_{\text{RT}} - 1\right)
241  * + \frac{4}{r^3 \omega_{\text{RT}}^2}\\
242  * U &\equiv U^A q_A = \frac{\eth \omega_{\text{RT}}}{r}.
243  * \f}
244  * and \f$q_A\f$ is the angular dyad on the unit sphere. The Robinson-Trautman
245  * scalar \f$\omega_{\text{RT}}\f$ is independent of the Bondi radius \f$r\f$,
246  * so all radial derivatives of \f$\omega_{\text{RT}}\f$ have been dropped
247  */
248  void dr_spherical_metric(
249  gsl::not_null<
250  tnsr::aa<DataVector, 3, ::Frame::Spherical<::Frame::Inertial>>*>
251  dr_spherical_metric,
252  size_t l_max, double time) const noexcept override;

254  /*!
255  * \brief Compute time derivative of the spherical coordinate metric of the
256  * Robinson-Trautman solution generated by the time-evolved scalar
257  * \f$\omega_{\text{RT}}\f$.
258  *
259  * \details The time derivative of the Robinson-Trautman solution is obtained
260  * by differentiating the expressions from the documentation for
261  * `RobinsonTrautman::spherical_metric()`:
262  *
263  * \f{align*}{
264  * \partial_t g_{a b} dx^a dx^b = -( \partial_u \omega_{\text{RT}} (r W + 1)
265  * + \omega_{\text{RT}} \partial_u W
266  * - r^2 (\bar U \partial_u U + U \partial_u \bar U)) (dt - dr)^2
267  * - 2 \partial_u \omega_{\text{RT}} (dt - dr) dr
268  * - 2 r^2 \partial_u U^A q_{AB} dx^B (dt - dr),
269  * \f}
270  *
271  * where \f$q_{A B}\f$ represents the angular unit sphere metric, and the
272  * remaining Bondi-Sachs scalars and angular tensors are defined in terms of
273  * the Robinson-Trautman scalar \f$\omega_{\text{RT}}\f$
274  *
275  * \f{align*}{
276  * W &= \frac{1}{r}\left(\omega_{\text{RT}}
277  * + \eth \bar \eth \omega_{\text{RT}} - 1\right)
278  * - \frac{2}{r^2 \omega_{\text{RT}}^2}\\
279  * \partial_u W &= \frac{1}{r}\left(\partial_u \omega_{\text{RT}}
280  * + \eth \bar \eth \partial_u \omega_{\text{RT}}\right)
281  * + \frac{4 \partial_u \omega_{\text{RT}}}{r^2 \omega_{\text{RT}}^3} \\
282  * \partial_u U &= q_A \partial_u U^A = \frac{\eth \partial_u
283  * \omega_{\text{RT}}}{r}, \f}
284  *
285  * and \f$q_A\f$ is the angular dyad on the unit sphere; and the time
286  * derivative of the Robinson-Trautman scalar \f$\omega_{\text{RT}}\f$ is
287  *
288  * \f[
289  * \partial_u \omega_{\text{RT}} =
290  * \frac{1}{12} \left(-\omega^4_{\text{RT}} \eth^2 \bar \eth^2
291  * \omega_{\text{RT}} + \omega_{\text{RT}}^3 (\eth^2 \omega_{\text{RT}})
292  * (\bar \eth^2 \omega_{\text{RT}}) \right)
293  * \f]
294  */
